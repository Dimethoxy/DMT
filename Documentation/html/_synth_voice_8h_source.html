15class SynthVoice : public juce::SynthesiserVoice
16{
17
18public:
19 //============================================================================
20 bool canPlaySound(juce::SynthesiserSound* sound) override
21 {
22 return dynamic_cast<juce::SynthesiserSound*>(sound) != nullptr;
23 }
24 void controllerMoved(int controllerNumber, int newControllerValue) override {}
25 void pitchWheelMoved(int newPitchWheelValue) override {}
26 //============================================================================
27 void prepareToPlay(double sampleRate, int samplesPerBlock, int outputChannels)
28 {
29 // Exit if the sample rate is invalid
30 if (sampleRate <= 0)
31 return;
32
33 // Prepare envelope
34 gainEnvelope.setSampleRate(sampleRate);
35 pitchEnvelope.setSampleRate(sampleRate);
36
37 // Prepare oscillator
38 osc.setSampleRate(sampleRate);
39
40 // Set state as prepared
41 isPrepared = true;
42 }
43 //============================================================================
44 void startNote(int midiNoteNumber,
45 float velocity,
46 juce::SynthesiserSound* sound,
47 int currentPitchWheelPosition) override
48 {
49 // Set oscillator frequency
50 osc.setPhase(0.0f);
51 note = midiNoteNumber;
52 ;
53
54 // Start envelopes
55 setEnvelopes();
56 gainEnvelope.noteOn();
57 pitchEnvelope.noteOn();
58
59 callOnNoteReceivers();
60 }
61 void stopNote(float velocity, bool allowTailOff) override {}
62 //============================================================================
63 void renderNextBlock(juce::AudioBuffer<float>& outputBuffer,
64 int startSample,
65 int numSamples) override
66 {
67 // Exit if the voice is not playing or the voice is not prepared
68 if (!isVoiceActive() || !isPrepared)
69 return;
70
71 osc.setWaveformType(chainSettings->waveformType);
72 osc.setDrive(chainSettings->oscDrive);
73 osc.setBias(chainSettings->oscBias);
74 osc.setBend(chainSettings->oscBend);
75 osc.setPwm(chainSettings->oscPwm);
76 osc.setSync(chainSettings->oscSync);
77
78 auto endSample = numSamples + startSample;
79 auto* leftChannel = outputBuffer.getWritePointer(0);
80 auto* rightChannel = outputBuffer.getWritePointer(1);
81 for (int sample = startSample; sample < endSample; sample++) {
82 osc.setFrequency(getFrequency());
83 float currentSample = osc.getNextSample();
84 gain(currentSample);
85 leftChannel[sample] = currentSample;
86 rightChannel[sample] = currentSample;
87 }
88 }
89 //============================================================================
90 void setChainSettings(dmt::ChainSettings chainSettings)
91 {
92 this->chainSettings = std::make_unique<dmt::ChainSettings>(chainSettings);
93 };
94 //============================================================================
95 void addOnNoteReceivers(std::function<void()> callbackFunction)
96 {
97 onNoteReceivers.push_back(callbackFunction);
98 }
99 void callOnNoteReceivers()
100 {
101 for (std::function<void()> func : onNoteReceivers) {
102 func();
103 }
104 }
105 //============================================================================
106private:
107 std::unique_ptr<dmt::ChainSettings> chainSettings;
108 dmt::AnalogOscillator osc;
109 dmt::AhdEnvelope gainEnvelope;
110 dmt::AhdEnvelope pitchEnvelope;
111 int note = 0;
112 float pitchDepth = 0.7;
113 bool isPrepared = false;
114
115 std::vector<std::function<void()>> onNoteReceivers;
116 //============================================================================
117 float getFrequency()
118 {
119 int semitones = (8 * chainSettings->oscOctave) + chainSettings->oscSemitone;
120 float baseFreq = juce::MidiMessage::getMidiNoteInHertz(note + semitones);
121 float freqModDepth =
122 chainSettings->oscOctave * baseFreq + chainSettings->modDepth;
123 float envelopeSample = pitchEnvelope.getNextSample();
124 float newFreq =
125 juce::mapToLog10(envelopeSample, baseFreq, baseFreq+freqModDepth);
126 return std::clamp(newFreq, 20.0f, 20000.0f);
127 }
128
129 void gain(float& currentSample)
130 {
131 float envGain = gainEnvelope.getNextSample();
132 float oscGain =
133 juce::Decibels::decibelsToGain(chainSettings->oscGain, -96.0f);
134 currentSample = currentSample * envGain * oscGain;
135 }
136 //============================================================================
137 void setEnvelopes()
138 {
139 dmt::AhdEnvelope::Parameters envParameters;
140 envParameters.attack = chainSettings->ampAttack;
141 envParameters.hold = chainSettings->ampHold;
142 envParameters.decay = chainSettings->ampDecay;
143 envParameters.attackScew = 0;
144 envParameters.decayScew = 0;
145 gainEnvelope.setParameters(envParameters);
146 envParameters.attack = 0.0;
147 envParameters.hold = 0.0f;
148 envParameters.decay = chainSettings->modDecay;
149 envParameters.decayScew = chainSettings->modScew;
150 pitchEnvelope.setParameters(envParameters);
151 }
152 //============================================================================
153};
