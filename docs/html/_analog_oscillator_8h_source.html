127 inline const float getBendedPhase(float rawPhase) const noexcept
128 {
129 auto bendedPhase = rawPhase;
130
131 float positiveCycleSize = posityCycleRatio * twoPi;
132 float negativeCycleRatio = 1.0f - posityCycleRatio;
133 float negativeCycleSize = negativeCycleRatio * twoPi;
134
135 if (rawPhase <= positiveCycleSize) {
136 bendedPhase /= (posityCycleRatio * 2.0f);
137 }
138
139 if (rawPhase > positiveCycleSize) {
140 bendedPhase = (rawPhase - positiveCycleSize) / negativeCycleSize;
141 bendedPhase = bendedPhase * pi + pi;
142 }
143 return bendedPhase;
144 }

157 }
