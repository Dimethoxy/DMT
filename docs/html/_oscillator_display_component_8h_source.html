16class OscillatorDisplayComponent

20 using Settings = dmt::AppSettings::OscillatorDisplay;

23public:
24 //============================================================================
25 OscillatorDisplayComponent(juce::AudioProcessorValueTreeState& apvts)
26 : apvts(apvts)
27 , chainSettings(apvts)
28 {
29 outerShadow.radius = Settings::outerShadowRadius;
30 outerShadow.colour = Settings::outerShadowColour;
31 innerShadow.radius = Settings::innerShadowRadius;
32 innerShadow.colour = Settings::innerShadowColour;
33 osc.setSampleRate((float)resolution + 1.0f);
34 osc.setFrequency(1.0f);
35 updateDisplay(chainSettings);
36 startTimerHz(60);
37 }
38 //============================================================================
39 void paint(juce::Graphics& g) override
40 {
41 const auto bounds = this->getLocalBounds().toFloat();
42
43 g.setColour(Settings::backgroundColour);
44 g.fillEllipse(bounds);
45
46 const auto borderSize = Settings::borderSize;
47 const auto borderBounds = bounds.reduced(Settings::margin);
48 const auto innerBounds = borderBounds.reduced(borderSize);
49 const auto outerCornerSize = Settings::outerCornerSize;
50 const auto innerCornerSize = Settings::innerCornerSize;
51
52 if (Settings::drawOuterShadow) {
53 juce::Path outerShadowPath;
54 outerShadowPath.addEllipse(borderBounds);
55 outerShadow.drawOuterForPath(g, outerShadowPath);
56 }
57
58 g.setColour(Settings::foregroundColour);
59 g.fillEllipse(innerBounds);
60
61 if (Settings::drawInnerShadow) {
62 juce::Path innerShadowPath;
63 innerShadowPath.addEllipse(innerBounds);
64 innerShadow.drawInnerForPath(g, innerShadowPath);
65 }
66
67 auto graphPath = getPath(borderBounds.reduced(Settings::graphSize / 2.0f));
68 auto integralPath = graphPath;
69 integralPath.closeSubPath();
70
71 g.setColour(Settings::integralColour);
72 g.fillPath(integralPath);
73
74 g.setColour(Settings::graphColor);
75 g.strokePath(graphPath, juce::PathStrokeType(Settings::graphSize));
76
77 g.setColour(Settings::borderColour);
78 g.drawEllipse(borderBounds.reduced(Settings::borderSize / 2.0f),
79 Settings::borderSize);
80 }

83private:
84 dmt::Shadow outerShadow;
85 dmt::Shadow innerShadow;
86 dmt::Shadow lineShadow;
87
88 dmt::AnalogOscillator osc;
89 juce::dsp::LookupTable<float> table;
90 dmt::ChainSettings chainSettings;
91 juce::AudioProcessorValueTreeState& apvts;
92
93 //==============================================================================
94 void timerCallback()
95 {
96 dmt::ChainSettings newChainSettings(apvts);
97 if (isParametersChanged(newChainSettings)) {
98 chainSettings = newChainSettings;
99 osc.setWaveformType(chainSettings.waveformType);
100 osc.setDrive(chainSettings.oscDrive);
101 osc.setBias(chainSettings.oscBias);
102 osc.setBend(chainSettings.oscBend);
103 osc.setPwm(chainSettings.oscPwm);
104 osc.setSync(chainSettings.oscSync);
105 this->buildTable();
106 this->repaint();
107 }
108 }
109
110 void updateDisplay(dmt::ChainSettings& newChainSettings)
111 {
112 chainSettings = newChainSettings;
113 osc.setWaveformType(chainSettings.waveformType);
114 osc.setBend(chainSettings.oscBend);
115 osc.setPwm(chainSettings.oscPwm);
116 osc.setSync(chainSettings.oscSync);
117 this->buildTable();
118 this->repaint();
119 }
120
121 void buildTable()
122 {
123 osc.setPhase(0.0f);
124 table.initialise([&](size_t index) { return (float)osc.getNextSample(); },
125 resolution);
126 }
127
128 bool isParametersChanged(dmt::ChainSettings& newChainSettings)
129 {
130 bool waveformChanged =
131 chainSettings.waveformType != newChainSettings.waveformType;
132 bool driveChanged = chainSettings.oscDrive != newChainSettings.oscDrive;
133 bool biasChanged = chainSettings.oscBias != newChainSettings.oscBias;
134 bool bendChanged = chainSettings.oscBend != newChainSettings.oscBend;
135 bool pwmChanged = chainSettings.oscPwm != newChainSettings.oscPwm;
136 bool syncChanged = chainSettings.oscSync != newChainSettings.oscSync;
137 return waveformChanged || driveChanged || biasChanged || bendChanged ||
138 pwmChanged || syncChanged;
139 }
140
141 //==============================================================================
142 juce::Path getPath(juce::Rectangle<float> bounds)
143 {
144 bounds.setY(bounds.getY() + (bounds.getHeight() / 10.0f));
145 bounds.setHeight(bounds.getHeight() - (bounds.getHeight() / 5.0f));
146
147 auto outerBounds = bounds;
148 bounds = bounds.reduced(bounds.getWidth() / 6.0f);
149
150 juce::Path path;
151
152 auto startX = bounds.getX();
153 auto startY = bounds.getY() + (bounds.getHeight() / 2.0f);
154 juce::Point<float> start(startX, startY);
155 path.startNewSubPath(start);
156
157 auto width = bounds.getWidth();
158
159 for (int i = 0; i < width; i++) {
160 auto x = bounds.getX() + i;
161 auto y = bounds.getY() + (bounds.getHeight() / 2.0f) -
162 (table[i / width * resolution] * bounds.getHeight() / 2.0f);
163 juce::Point<float> p(x, y);
164 path.lineTo(p);
165 }
166
167 auto endX = bounds.getX() + bounds.getWidth();
168 auto endY = bounds.getY() + (bounds.getHeight() / 2.0f);
169 juce::Point<float> end(endX, endY);
170 path.lineTo(end);
171
172 return path;
173 }
